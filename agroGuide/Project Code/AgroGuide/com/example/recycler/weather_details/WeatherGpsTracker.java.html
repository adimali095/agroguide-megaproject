55       public Location getLocation() {
56           try {
57               locationManager = (LocationManager) mContext.getSystemService(LOCATION_SERVICE);
58   
59               // getting GPS status
60               isGPSEnabled = locationManager.isProviderEnabled(LocationManager.GPS_PROVIDER);
61   
62               // getting network status
63               isNetworkEnabled = locationManager.isProviderEnabled(LocationManager.NETWORK_PROVIDER);
64   
65               if (!isGPSEnabled && !isNetworkEnabled) {
66                   // no network provider is enabled
67   
68               } else {
69                   this.canGetLocation = true;
70                   // First get location from Network Provider
71                   if (isNetworkEnabled) {
72                       //check the network permission
73   
74                       if (ActivityCompat.checkSelfPermission(mContext, Manifest.permission.ACCESS_FINE_LOCATION) != PackageManager.PERMISSION_GRANTED && ActivityCompat.checkSelfPermission(mContext, Manifest.permission.ACCESS_COARSE_LOCATION) != PackageManager.PERMISSION_GRANTED) {
75                           ActivityCompat.requestPermissions((Activity) mContext, new String[]{android.Manifest.permission.ACCESS_FINE_LOCATION, Manifest.permission.ACCESS_COARSE_LOCATION}, 101);
76                       }
77                       locationManager.requestLocationUpdates(
78                               LocationManager.NETWORK_PROVIDER,
79                               MIN_TIME_BW_UPDATES,
80                               MIN_DISTANCE_CHANGE_FOR_UPDATES, this);
81   
82                       Log.d("Network", "Network");
83                       if (locationManager != null) {
84                           location = locationManager
85                                   .getLastKnownLocation(LocationManager.NETWORK_PROVIDER);
86   
87                           if (location != null) {
88                               latitude = location.getLatitude();
89                               longitude = location.getLongitude();
90                           }
91                       }
92                   }
93   
94                   // if GPS Enabled get lat/long using GPS Services
95                   if (isGPSEnabled) {
96                       if (location == null) {
97                           //check the network permission
98                           if (ActivityCompat.checkSelfPermission(mContext, Manifest.permission.ACCESS_FINE_LOCATION) != PackageManager.PERMISSION_GRANTED && ActivityCompat.checkSelfPermission(mContext, Manifest.permission.ACCESS_COARSE_LOCATION) != PackageManager.PERMISSION_GRANTED) {
99                               ActivityCompat.requestPermissions((Activity) mContext, new String[]{android.Manifest.permission.ACCESS_FINE_LOCATION, Manifest.permission.ACCESS_COARSE_LOCATION}, 101);
100                          }
101                          locationManager.requestLocationUpdates(
102                                  LocationManager.GPS_PROVIDER,
103                                  MIN_TIME_BW_UPDATES,
104                                  MIN_DISTANCE_CHANGE_FOR_UPDATES, this);
105  
106                          Log.d("GPS Enabled", "GPS Enabled");
107                          if (locationManager != null) {
108                              location = locationManager
109                                      .getLastKnownLocation(LocationManager.GPS_PROVIDER);
110                              if (location != null) {
111                                  latitude = location.getLatitude();
112                                  longitude = location.getLongitude();
113                              }
114                          }
115                      }
116                  }
117              }
118          } catch (Exception e) {
119              e.printStackTrace();
120          }
121  
122          return location;
123      }
