96       public void callProfile(View view) {
97           String logUsername = getIntent().getStringExtra("username");
98           String logEmail = getIntent().getStringExtra("email");
99           String logPhoneNo = getIntent().getStringExtra("phoneNo");
100          String logFullname = getIntent().getStringExtra("fullname");
101          String logPasswrod = getIntent().getStringExtra("password");
102  
103          Intent intent = new Intent(getApplicationContext(),UserProfile.class);
104          intent.putExtra("fullname",logFullname);
105          intent.putExtra("username",logUsername);
106          intent.putExtra("email",logEmail);
107          intent.putExtra("password",logPasswrod);
108          intent.putExtra("phoneNo",logPhoneNo);
109  
110          startActivity(intent);
111      }
